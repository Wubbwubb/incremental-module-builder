22  import java.util.LinkedList;
23  import java.util.List;
24  import java.util.Objects;
25  import java.util.concurrent.ExecutionException;
26  
27  import org.apache.maven.execution.MavenSession;
28  import org.apache.maven.execution.ProjectDependencyGraph;
29  import org.apache.maven.lifecycle.internal.LifecycleModuleBuilder;
30  import org.apache.maven.lifecycle.internal.ReactorContext;
31  import org.apache.maven.lifecycle.internal.TaskSegment;
32  import org.apache.maven.project.MavenProject;
33  import org.slf4j.Logger;
34  import org.slf4j.LoggerFactory;
35  
36  /**
37   * @author Karl Heinz Marbaise <khmabaise@apache.org>
38   */
39  class IncrementalModuleBuilderImpl {
40  
41      private final Logger logger = LoggerFactory.getLogger(getClass());
42  
43      private MavenSession mavenSession;
44      private List<MavenProject> projects;
45      private List<TaskSegment> taskSegments;
46      private ReactorContext reactorContext;
47  
48      private final LifecycleModuleBuilder lifecycleModuleBuilder;
49  
50      IncrementalModuleBuilderImpl(List<MavenProject> selectedProjects, LifecycleModuleBuilder lifecycleModuleBuilder,
51  	    MavenSession session, ReactorContext reactorContext, List<TaskSegment> taskSegments) {
52  
53  	this.lifecycleModuleBuilder = Objects.requireNonNull(lifecycleModuleBuilder,
54  		"lifecycleModuleBuilder is not allowed to be null.");
55  	this.mavenSession = Objects.requireNonNull(session, "session is not allowed to be null.");
56  	this.taskSegments = Objects.requireNonNull(taskSegments, "taskSegements is not allowed to be null");
57  	this.reactorContext = Objects.requireNonNull(reactorContext, "reactorContext is not allowed to be null.");
58  
59  	ProjectDependencyGraph projectDependencyGraph = session.getProjectDependencyGraph();
60  
61  	List<MavenProject> intermediateResult = new LinkedList<>();
62  
63  	for (MavenProject selectedProject : selectedProjects) {
64  	    intermediateResult.add(selectedProject);
65  	    // Up or downstream ? (-amd)
66  	    intermediateResult.addAll(projectDependencyGraph.getDownstreamProjects(selectedProject, true));
67  	    // TODO: Need to think about this? -am ?
68  	    // intermediateResult.addAll(projectDependencyGraph.getUpstreamProjects(selectedProject,
69  	    // true));
70  	}
71  
72  	List<MavenProject> result = new LinkedList<>();
73  
74  	for (MavenProject project : intermediateResult) {
75  	    if (!result.contains(project)) {
76  		result.add(project);
77  	    }
78  	}
79  
80  	this.projects = result;
81  
82      }
83  
84      public void build() throws ExecutionException, InterruptedException {
85  	this.mavenSession.setProjects(this.projects);
86  
87  	logger.info("New Calculated Reactor:");
88  	for (MavenProject mavenProject : this.mavenSession.getProjects()) {
89  	    logger.info(" {}", mavenProject.getName());
90  	}
91  
92  	for (TaskSegment taskSegment : this.taskSegments) {
93  	    logger.debug("Segment");
94  	    List<Object> tasks = taskSegment.getTasks();
95  	    for (Object task : tasks) {
96  		logger.debug(" Task:" + task);
97  	    }
98  	    for (MavenProject mavenProject : mavenSession.getProjects()) {
99  		logger.info("Building project: {}", mavenProject.getId());
100 		lifecycleModuleBuilder.buildProject(mavenSession, reactorContext, mavenProject, taskSegment);
101 	    }
102 	}
103     }
